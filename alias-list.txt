### ALIAS LIST ###

# Basic alias #
alias ls="ls --color=auto"
alias diroot="sudo nautilus"
alias chmod="chmod -R"
alias aliasls="cat ~/Documents/alias"

# Git alias #
alias gitnote="nano ~/Documents/git"
alias gits="git status"

# Web alias #

alias sql="mysql -u root -p"

alias mongod="service mongod start"

# NodeJS alias #
